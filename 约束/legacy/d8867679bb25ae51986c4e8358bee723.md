### 4. 配额超额处理策略（fail fast + 记录）

```python
# 配额管理器
class QuotaManager:
    def __init__(self, quota_allocation: dict):
        self.quotas = quota_allocation
        self.usage = {op: 0 for op in quota_allocation}
        self.total_generated = 0
    
    def check_and_allocate(self, operator_id: str) -> bool:
        """检查配额并分配，超额时fail fast"""
        current_ratio = self.usage[operator_id] / self.total_generated if self.total_generated > 0 else 0
        max_quota = self.quotas[operator_id]["max_quota"]
        
        if current_ratio >= max_quota:
            return False  # 配额已满，fail fast
        
        self.usage[operator_id] += 1
        self.total_generated += 1
        return True
    
    def get_quota_report(self) -> dict:
        """生成配额使用报告"""
        return {
            "usage_by_operator": {
                op: {
                    "count": self.usage[op],
                    "ratio": self.usage[op] / self.total_generated if self.total_generated > 0 else 0,
                    "quota_limit": self.quotas[op]["max_quota"],
                    "quota_exhausted": (self.usage[op] / self.total_generated if self.total_generated > 0 else 0) >= self.quotas[op]["max_quota"]
                }
                for op in self.quotas
            },
            "total_generated": self.total_generated,
            "overall_compliance": "pass" if all(
                (self.usage[op] / self.total_generated if self.total_generated > 0 else 0) <= self.quotas[op]["max_quota"] 
                for op in self.quotas
            ) else "fail"
        }
```
